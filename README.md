prompt
======

修改默认的prompt

PROMPT.init({
  container:'',	//最外层的选择器
	header:'',		//提示头
	text:'',		//显示文本
	yesButton:'',	//确定按钮显示文本
	noButton:'',	//取消按钮显示文本		
	headerTarget:'',//提示头的选择器
	textTarget:'',	//内容选择器
	yesTarget:'',	//确定按钮选择器
	noTarget:''		//取消按钮选择器
	yesButtonDisplay:'',//是否显示确定按钮
	noButtonDisplay:'',//是否显示取消按钮
	yesCallback:'',		//确定按钮的回调
	noCallback''		//取消按钮的回调
});
